301  std::vector < T> result;
302  int counter = 0;
303  if(i >= (int)source.size() || j < 0) return result;
304

307  // check your controlling indices
308  int maxElement = source.size() -1;
309  i = std::max(std::min(i,maxElement), 0);

314  result[counter] = source[k];
